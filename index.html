Perbaikan Fungsi Muat Data Saat Aplikasi Dibuka window.onload = function() { // Ambil data dari HP if (localStorage.getItem('myDailyGrowthState')) { try { const savedState = JSON.parse(localStorage.getItem('myDailyGrowthState')); // Amankan agar item spiritual tidak menjadi kosong/hilang if (savedState.spiritualItems && savedState.spiritualItems.length > 0) { appState = Object.assign({}, appState, savedState); } else { // Jika data di HP rusak/kosong, tetap pakai data default bawaan delete savedState.spiritualItems; appState = Object.assign({}, appState, savedState); } } catch (e) { console.error("Gagal memuat memori HP", e); } } // Set tanggal real-time const options = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' }; document.getElementById('current-date').innerText = new Date().toLocaleDateString('id-ID', options); // Render semua elemen ke layar initChart(); renderAll(); switchTab(appState.currentTab); updatePuasaReminder(); // Jalankan sistem waktu sholat yang anti-bug updateCountdownSholat(); setInterval(updateCountdownSholat, 60000); }; // 2. Perbaikan Sistem Jadwal Sholat (Anti-Blank / Anti-Ngebug) function updateCountdownSholat() { const dummyPrayers = [ { name: 'Subuh', time: '04:47' }, { name: 'Dzuhur', time: '12:05' }, { name: 'Ashar', time: '15:26' }, { name: 'Maghrib', time: '17:59' }, { name: 'Isya', time: '19:12' } ]; const now = new Date(); const currentMinutes = now.getHours() * 60 + now.getMinutes(); let nextPrayer = null; // Cari sholat yang jamnya di atas jam sekarang for (let i = 0; i < dummyPrayers.length; i++) { const p = dummyPrayers[i]; const timeParts = p.time.split(':'); const h = parseInt(timeParts[0], 10); const m = parseInt(timeParts[1], 10); if ((h * 60 + m) > currentMinutes) { nextPrayer = p; break; } } // Jika sudah lewat Isya, otomatis set ke Subuh besok if (!nextPrayer) { nextPrayer = dummyPrayers[0]; } // Amankan elemen HTML agar tidak blank const nameEl = document.getElementById('next-prayer-name'); const timeEl = document.getElementById('next-prayer-time'); if (nameEl && timeEl) { nameEl.innerText = nextPrayer.name; timeEl.innerText = nextPrayer.time + " WIB"; } }
